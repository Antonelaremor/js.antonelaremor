// As patricinhas de Beverly Hills, 12, comédia romântica // De repente 30, livre, comédia-romance // Harry Potter e o Prisioneiro de Azkaban, livre, infantil-fantasia // Meu malvado favorito, livre, infantil-comédia // Frozen - uma aventura congelante, livre, infantil-musical let campoIdade; let campoFantasia; let campoComedia; function setup() { createCanvas(400, 400); createElement ("h2", "Recomendador de filmes"); createSpan ("Sua idade:"); campoIdade = createInput ("15"); campoFantasia = createCheckbox ("Gosta fantasia?") campoComedia = createCheckbox ("Gosta de Comédia"); } function draw() { background(220); let idade = campoIdade.value (); let gostaDeFantasia = campoFantasia.checked(); let gostaDeComedia = campoComedia.checked(); let recomendacao = geraRecomendacao(idade, gostaDeFantasia, gostaDeComedia); text (recomendacao, width / 2, height / 2); fill(color(76, 0, 115)); textAlign(CENTER, CENTER); textSize(25); text(recomendacao, width / 2, height / 2); } function geraRecomendacao (idade, gostaDeFantasia) { if (idade >= 10) { if (idade>= 12) { return " As patricinhas de Beverly Hills"; } else { if (gostaDeFantasia){ return "Harry Potter e o Prisioneiro de Azkaban"; } else { return " Meu malvado favorito"; } } } else { if (gostaDeFantasia) { return "Frozen - uma aventura congelante"; } else { return "De repente 30"; } } }
